#include <bits/stdc++.h>
using namespace std;
#define SIZE 11
double input[] = {0.99,0.19,0.15,0.13,0.21,0.36,0.45,0.50,0.98,0.75,0.01};


void display(vector<double>& v){
    vector<double>::iterator it = v.begin();
    while(it!=v.end()){
        cout << *(it) << " ";
        it++;
    }
}

void sorting(vector<double>& v){
    sort(v.begin(),v.end());
}
int main()
{
    vector< vector<double> > v(10);
    for(int i = 0 ; i < SIZE; i++)
    {
        double temp = input[i]*10;
        temp = floor(temp);
        int temp2 = temp;
        v[temp2].push_back(input[i]);
        cout << "Index = " << temp2 << " value = " << input[i] << endl;
    }

    for(int i = 0 ; i < 10 ;i++)
        sorting(v[i]);

    for(int i = 0 ; i < 10 ;i++)
        display(v[i]);

    return 0;
}
