#include <bits/stdc++.h>

using namespace std;

void merging(int low, int mid, int high, int n, vector<int>& arr){
    int l1,l2,i;
    vector<int> arr2(n);
    l1 = low;
    l2 = mid+1;
    i = low;
    for(;l1<=mid && l2<= high;i++){
        if( arr.at(l1) <arr.at(l2) )
            arr2.at(i) = arr.at(l1++);
        else
            arr2.at(i) = arr.at(l2++);
    }

    while(l1<=mid)
        arr2.at(i++) = arr.at(l1++);

    while(l2<=high)
        arr2.at(i++) = arr.at(l2++);

    for(i=low;i<=high;i++)
        arr.at(i)  = arr2.at(i);

}

void sort_(int low, int high, int n, vector<int>& arr){
    int mid = (low + high) / 2;
    if(low < high){
        sort_(low,mid, n, arr);
        sort_(mid+1,high, n, arr);
        merging(low,mid,high, n, arr);
    }else
        return;

}

int main(){


    int arr[12] = {144,255,222,333,11,2,3,4,0,5,6,7};
    vector<int> arr2(arr,arr+12);
    sort_(0,11, 12, arr2);

    for(int i = 0 ; i < 12 ; i++)
        cout << arr2.at(i) << " ";


    return 0;
}
